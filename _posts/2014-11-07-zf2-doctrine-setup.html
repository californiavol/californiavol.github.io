ZF2 Doctrine Setup

MacOSX Yosemite Terminal Command Line

	
git clone git://github.com/zendframework/ZendSkeletonApplication.git new-app-dir

cd new-app-dir

php composer.phar self-update
php composer.phar install

php composer.phar require zendframework/zftool:dev-master

php composer.phar require zendframework/zend-developer-tools:dev-master

cp vendor/zendframework/zend-developer-tools/config/zenddevelopertools.local.php.dist config/autoload/zenddevelopertools.local.php

//allows the usage of the (Zend) Db collector
//https://github.com/bjyoungblood/BjyProfiler
$ php composer.phar require bjyoungblood/bjy-profiler:dev-master


//OcraServiceManager can help you track dependencies within your application
//https://github.com/Ocramius/OcraServiceManager
$ php composer.phar require ocramius/ocra-service-manager


//doctrine 2
php composer.phar require doctrine/doctrine-orm-module:0.*
